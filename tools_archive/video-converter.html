🛠 ffmpeg 常用转换命令

# MP4 → WebM
ffmpeg -i input.mp4 -c:v libvpx-vp9 -crf 30 -b:v 0 output.webm
# 压缩视频 (调整crf: 18-28)
ffmpeg -i input.mp4 -c:v libx264 -crf 23 output.mp4
# 提取音频
ffmpeg -i input.mp4 -q:a 0 output.mp3
